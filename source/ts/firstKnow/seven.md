```ts
class Person {
    private _name: string;
    constructor(name: string) {
        this._name = name;
    }
    get name():string {
        return this._name;
    }
    set name(newName:string) {
        if (newName.length < 8) {
            throw new Error('名字最少8位');
        } else {
            this._name = newName;
        }
    }
    show(): void {
        console.log(this._name);
    }
}
let p:Person=new Person('huang');
p.show();//huang
p.name="huangjmlsadjf";
console.log(p.name);//huangjmlsadjf
```
